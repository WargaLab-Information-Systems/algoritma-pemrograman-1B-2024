SET skorjaka = 1100
SET ipkjaka = 3.5
SET skorida = 1000
SET ipkida = 3.5

    PRINT "jaka lulus"

    PRINT "jaka tidak lulus"

IF skorida >= 1000 AND ipkida >= 3 THEN
    PRINT "Ida lulus"
ELSE
